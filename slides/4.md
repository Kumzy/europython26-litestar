@get("/orders", guards=[require_user])
async def list_orders(
    user: NamedDependency[User],
    db: NamedDependency[Session],
    page: FromQuery[int] = 1,
    search: FromQuery[str | None] = None,
) -> list[OrderDTO]:
    orders = await query_orders(db, page=page, search=search)
    await queue.enqueue("process_orders", user.id)
    return OrderDTO.from_models(orders)
